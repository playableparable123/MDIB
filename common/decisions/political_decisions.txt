			if = {
				limit = {
					has_government = western_democracy
				}

					var = civil_war_size_var
					value = party_popularity@western_democracy
				}
			}
			if = {
				limit = {
					has_government = marxism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@marxism
				}
			}
			if = {
				limit = {
					has_government = institutionalism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@institutionalism
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = new_right
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_fascism_civil_war
			clr_country_flag = military_support_for_fascism_civil_war
